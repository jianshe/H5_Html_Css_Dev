var setting = {
  //目标 action sheet, 可为 Zepto 对象或者 css 选择器
  target: null,

  //被选中项高亮类名
  activeClass: 'active',

  //选择单个项目后的回调
  onSelect: function(){},

  //取消选择的回调
  cancel: function(){}
};
$('[data-action-sheet="style1"]').actionSheet({
  target: '.login-timeset-pop',
  onSelect: function(value){
    $.tips({
      content: '您选择了: ' + value
    });
  },
  cancel: function(){
    $.tips({
      content: '您取消了选择'
    });
  }
});
