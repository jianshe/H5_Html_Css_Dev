});

*/


调用弹窗

EM.util.emH5Alert({
  content: '要换背景色吗？',
  okCallbackName: 'onAlertOk'
});
window.onAlertOk = function(){
  document.body.style.background = 'blue';
};
/**
* 调用系统原生 alert
* @param option
* option.content 弹出内容，必须
* option.title 弹框标题，默认“东方财富网”
* option.okLabel 确认按钮的文字，默认“确定”
* option.okCallbackName 点击确认按钮的回调，如果指定了方法名，则必须全局定义该方法
* option.cancelLabel 取消按钮的文字，默认“取消”
* option.cancelCallbackName 点击取消按钮的回调，如果指定了方法名，则必须全局定义该方法
*/
