action-sheet2插件 返回

$('#sheet1').actionSheet2({
    className: 'login-timeset-pop',
    data: [
        {label: '照相', value: 1},
        {label: '从相册中选择', value: 2}
    ],
    onSelect: function(value){
        console.log('sheet1:' + value);
    },
    onCancel: function(){
        console.log('取消 sheet1');
    }
});
